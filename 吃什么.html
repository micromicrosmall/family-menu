// ==================== 满意度记录系统 ====================

class SatisfactionTracker {

    constructor() {

        this.records = JSON.parse(localStorage.getItem('menuSatisfaction') || '[]');

        this.monthlyStats = JSON.parse(localStorage.getItem('monthlyStats') || '{}');

    }

    

    record(mealType, members, menu, satisfied) {

        const record = {

            date: new Date().toISOString().split('T')[0],

            meal: mealType,

            members: members,

            menu: menu,

            satisfied: satisfied,

            timestamp: Date.now()

        };

        

        this.records.push(record);

        

        // 只保留最近1000条记录

        if (this.records.length > 1000) {

            this.records = this.records.slice(-1000);

        }

        

        // 更新月度统计

        const month = new Date().toISOString().slice(0, 7); // YYYY-MM

        if (!this.monthlyStats[month]) {

            this.monthlyStats[month] = {

                total: 0,

                satisfied: 0,

                byMeal: {},

                byMember: {}

            };

        }

        

        this.monthlyStats[month].total++;

        if (satisfied) this.monthlyStats[month].satisfied++;

        

        // 按餐别统计

        if (!this.monthlyStats[month].byMeal[mealType]) {

            this.monthlyStats[month].byMeal[mealType] = { total: 0, satisfied: 0 };

        }

        this.monthlyStats[month].byMeal[mealType].total++;

        if (satisfied) this.monthlyStats[month].byMeal[mealType].satisfied++;

        

        // 按成员统计

        members.forEach(member => {

            if (!this.monthlyStats[month].byMember[member]) {

                this.monthlyStats[month].byMember[member] = { total: 0, satisfied: 0 };

            }

            this.monthlyStats[month].byMember[member].total++;

            if (satisfied) this.monthlyStats[month].byMember[member].satisfied++;

        });

        

        this.save();

    }

    

    getSatisfactionRate() {

        if (this.records.length === 0) return 0;

        const satisfied = this.records.filter(r => r.satisfied).length;

        return Math.round((satisfied / this.records.length) * 100);

    }

    

    getMonthlyReport(month) {

        return this.monthlyStats[month] || null;

    }

    

    save() {

        localStorage.setItem('menuSatisfaction', JSON.stringify(this.records));

        localStorage.setItem('monthlyStats', JSON.stringify(this.monthlyStats));

    }

}

const tracker = new SatisfactionTracker();

let selectedMembers = [];

let currentMeal = '';

let currentMenu = '';

let currentDayIndex = 0;


// 初始化

document.addEventListener('DOMContentLoaded', function() {

    // 计算当前是几号（1-30），用于30天轮换

    const today = new Date();

    currentDayIndex = (today.getDate() - 1) % CONFIG.monthDays;

    

    // 绑定事件

    document.getElementById('startBtn').addEventListener('click', () => showStep(2));

    

    document.querySelectorAll('.member-btn').forEach(btn => {

        btn.addEventListener('click', function() {

            const member = this.getAttribute('data-member');

            toggleMember(member);

        });

    });

    

    document.getElementById('nextBtn').addEventListener('click', () => showStep(3));

    document.getElementById('backBtn1').addEventListener('click', () => showStep(1));

    document.getElementById('backBtn2').addEventListener('click', () => showStep(2));

    document.getElementById('backBtn3').addEventListener('click', resetSelection);

    

    document.querySelectorAll('.meal-btn').forEach(btn => {

        btn.addEventListener('click', function() {

            currentMeal = this.getAttribute('data-meal');

            showStep(4);

            generateMenu();

        });

    });

    

    document.getElementById('likeBtn').addEventListener('click', () => {

        recordSatisfaction(true);

        alert('✅ 已记录您的选择！\n当前满意度：' + tracker.getSatisfactionRate() + '%');

        resetToStart();

    });

    

    document.getElementById('dislikeBtn').addEventListener('click', () => {

        document.getElementById('changeBtn').style.display = 'block';

        currentDayIndex = (currentDayIndex + 1) % CONFIG.monthDays; // 换到下一天

        generateMenu();

    });

    

    document.getElementById('changeBtn').addEventListener('click', () => {

        currentDayIndex = (currentDayIndex + 1) % CONFIG.monthDays;

        generateMenu();

    });

    

    // 初始化显示

    updateSelectedCount();

});


// 显示指定步骤

function showStep(stepNumber) {

    document.querySelectorAll('.step').forEach(step => step.classList.remove('active'));

    document.getElementById(`step${stepNumber}`).classList.add('active');

    

    if (stepNumber === 2) {

        updateSelectedCount();

    }

}


// 切换成员选择

function toggleMember(member) {

    const index = selectedMembers.indexOf(member);

    if (index === -1) {

        selectedMembers.push(member);

    } else {

        selectedMembers.splice(index, 1);

    }

    

    // 更新按钮状态

    document.querySelectorAll('.member-btn').forEach(btn => {

        if (btn.getAttribute('data-member') === member) {

            btn.classList.toggle('selected');

        }

    });

    

    updateSelectedCount();

    document.getElementById('nextBtn').disabled = selectedMembers.length === 0;

}


// 更新选择人数显示

function updateSelectedCount() {

    const display = document.getElementById('selectedCount');

    if (selectedMembers.length === 0) {

        display.textContent = '请选择吃饭的人';

        display.style.color = '#999';

    } else {

        display.textContent = `已选 ${selectedMembers.length} 人：${selectedMembers.join('、')}`;

        display.style.color = '#667eea';

        

        // 显示健康提醒

        let healthTips = [];

        selectedMembers.forEach(member => {

            const issues = HEALTH_REQUIREMENTS[member].issues.slice(0, 2).join('、');

            healthTips.push(`${member}：${issues}`);

        });

    }

}


// 生成融合菜单

function generateMenu() {

    if (selectedMembers.length === 0 || !currentMeal) return;

    

    // 更新标题

    const title = document.getElementById('resultTitle');

    title.textContent = `${selectedMembers.join('、')}的${currentMeal}`;

    

    let menuItems = [];

    let healthNotes = [];

    

    // 根据餐别和人数生成菜单

    if (currentMeal === '早餐' || currentMeal === '加餐') {

        // 每人单独一份

        selectedMembers.forEach(member => {

            const recipes = MONTHLY_RECIPES[currentMeal][member];

            const recipe = recipes[currentDayIndex];

            menuItems.push(`🍽️ ${member}：${recipe}`);

            

            // 收集健康提醒

            const restrictions = HEALTH_REQUIREMENTS[member].restrictions.slice(0, 2);

            healthNotes.push(`${member}：${restrictions.join('，')}`);

        });

    } else {

        // 午餐和晚餐：融合菜单

        const peopleCount = selectedMembers.length;

        

        // 根据人数决定菜品数量

        let dishCount = 0;

        if (peopleCount === 1) {

            dishCount = 2; // 1人：两菜一汤

        } else if (peopleCount === 2) {

            dishCount = 3; // 2人：三菜一汤

        } else {

            dishCount = peopleCount + 1; // 每多一人加一个菜

        }

        

        // 添加汤

        const soups = MONTHLY_RECIPES[currentMeal]['汤'];

        const soup = soups[currentDayIndex];

        menuItems.push(`🍲 汤：${soup}`);

        

        // 添加主菜

        const dishesAdded = new Set();

        for (let i = 1; i < dishCount; i++) {

            // 轮流从不同成员中选择菜品

            const memberIndex = (i - 1) % selectedMembers.length;

            const member = selectedMembers[memberIndex];

            const memberRecipes = MONTHLY_RECIPES[currentMeal][member];

            const dish = memberRecipes[currentDayIndex];

            

            // 避免重复

            if (!dishesAdded.has(dish)) {

                menuItems.push(`🍛 ${dish}`);

                dishesAdded.add(dish);

                

                // 收集健康提醒

                const needs = HEALTH_REQUIREMENTS[member].issues.slice(0, 2);

                if (!healthNotes.some(note => note.includes(member))) {

                    healthNotes.push(`${member}：${needs.join('，')}`);

                }

            } else {

                i--; // 重新选择

            }

        }

    }

    

    // 保存当前菜单

    currentMenu = menuItems.join('\n');

    

    // 显示菜单

    document.getElementById('menuText').textContent = menuItems.join('\n');

    

    // 显示健康提醒

    const healthNote = document.getElementById('healthNote');

    healthNote.textContent = `健康提示：${[...new Set(healthNotes)].join('；')}`;

    

    // 隐藏换一个按钮

    document.getElementById('changeBtn').style.display = 'none';

}


// 记录满意度

function recordSatisfaction(satisfied) {

    tracker.record(currentMeal, selectedMembers, currentMenu, satisfied);

}


// 重置到开始

function resetToStart() {

    selectedMembers = [];

    currentMeal = '';

    document.querySelectorAll('.member-btn').forEach(btn => btn.classList.remove('selected'));

    showStep(1);

}


// 重置选择

function resetSelection() {

    showStep(2);

}
